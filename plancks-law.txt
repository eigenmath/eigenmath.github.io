-- Draw a graph of black-body radiation for 3000 Kelvin

-- x axis is frequency

-- y axis is joules per meter

T = 3000 kelvin

c = 299792458 meter/second
h = 6.62607004 10^(-34) kilogram meter^2 / second
k = 1.38064852 10^(-23) meter^2 kilogram second^(-2) kelvin^(-1)
joule = kilogram meter / second^2

rho = (2 h nu^3 / c^2) / (exp(h nu / (k T)) - 1)

f(x) = eval(rho,nu,x/second) meter / joule

xrange = (0,10^15)
yrange = (0,10^(-8))

draw(f)
